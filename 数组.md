fmt.Println(slice4)

// 访问切片
slice4[1:4] // 访问索引1-3的数据
 
slice4[:3]  // 访问索引0-2的数据

slice4[4:]  // 访问索引4到最后的数据

// 长度和大小
len(slice4)
cap(slice4)

// 追加
slice1 = append(slice1, 0) // 把空切片追加到slice1
new_slice = append(slice1, 2,3,4) // 将2,3,4追加到slice1

// 拷贝
copy(slice2,slice1) //copy slice1 到 slice2
